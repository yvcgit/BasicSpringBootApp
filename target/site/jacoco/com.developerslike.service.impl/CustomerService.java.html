package com.developerslike.service.impl;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

import org.apache.log4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.dao.DataAccessResourceFailureException;
import org.springframework.stereotype.Service;

import com.developerslike.dao.ICustomerDao;
import com.developerslike.data.CustomerData;
import com.developerslike.entity.Customer;
import com.developerslike.service.ICustomerService;

@Service
public class CustomerService implements ICustomerService{
	/**
	 * this serivce is used in customerDetailsapi which is serialized
	 * so we have to do service also serializable
	 */
	private static final long serialVersionUID = 1L;
	
	@Autowired
	private static Logger logger;
	@Autowired
	ICustomerDao customerDao;

	@Override
	public void saveCustomer(CustomerData customerData) {
		//logger.info("customerData  " + customerData);
		customerDao.save(toEntity(customerData));
	}

	private Customer toEntity(CustomerData customerData) {
		Customer customer = new Customer();
		customer.setAge(customerData.getAge());
		customer.setContactNumber(customerData.getContactNumber());
		customer.setEmail(customerData.getEmail());
		customer.setFirstName(customerData.getFirstName());
		customer.setLastName(customerData.getLastName());
		customer.setMiddleName(customerData.getMiddleName());
		//logger.info("customer  " + customer);

		return customer;
	}

	@Override
	public String updateCustomer(CustomerData customerData) {
		Optional<Customer> findByCustomerId = customerDao.findByCustomerId(customerData.getCustomerId());
		if (findByCustomerId.isPresent()) {
			customerDao.save(update(findByCustomerId.get(), customerData));
			return "update " + findByCustomerId.get().getFirstName() + " to " + customerData.getFirstName();
		}

		return "no record to update";
	}

	private Customer update(Customer customer, CustomerData customerData) {
		customer.setAge(customerData.getAge() > 0 ? customerData.getAge() : customer.getAge());
		customer.setContactNumber(customerData.getContactNumber() != null ? customerData.getContactNumber()
				: customer.getContactNumber());
		customer.setEmail(customerData.getEmail() != null ? customerData.getEmail() : customer.getEmail());
		customer.setFirstName(
				customerData.getFirstName() != null ? customerData.getFirstName() : customer.getFirstName());
		customer.setLastName(customerData.getLastName() != null ? customerData.getLastName() : customer.getLastName());
		customer.setMiddleName(
				customerData.getLastName() != null ? customerData.getLastName() : customer.getLastName());

		return customer;
	}

	@Override
	public CustomerData getCustomerById(long id) {
		//logger.debug("custome id = " + id);
		Optional<Customer> findById = customerDao.findByCustomerId(id);
		//logger.info("optional customer " + findById);
		if (!findById.isPresent())
			throw new DataAccessResourceFailureException("no result=====");
		return toData(findById.get());
	}

	private CustomerData toData(Customer customer) {
		CustomerData customerData = new CustomerData();
		customerData.setAge(customer.getAge());
		customerData.setContactNumber(customer.getContactNumber());
		customerData.setEmail(customer.getEmail());
		customerData.setFirstName(customer.getFirstName());
		customerData.setLastName(customer.getLastName());
		customerData.setMiddleName(customer.getMiddleName());
		customerData.setCustomerId(customer.getCustomerId());
		//logger.info(" customerData === " + customerData);
		return customerData;
	}

	@Override
	
	public String deleteCustomerById(long customerId) {
		try {
			customerDao.deleteById(Long.valueOf(customerId));
			return "Deleted";
		} catch (Exception e) {
			return e.getMessage();
		}
	}

	@Override
	public List<CustomerData> getCustomerList() {
		/**
		 * Replace the type specification in this constructor call with the diamond operator ("<>").
		 * Java 7 introduced the diamond operator (<>) to reduce the verbosity of generics code. 
		 * For instance, instead of having to declare a List's type in both its declaration and its constructor,
		 * you can now simplify the constructor declaration with <>, and the compiler will infer the type.
		 */
		//List<CustomerData> customerDataList = new ArrayList<CustomerData>();
		List<CustomerData> customerDataList = new ArrayList<>();
		Iterable<Customer> findAll = customerDao.findAll();
		//logger.info("list ====" + findAll.iterator().hasNext());
		findAll.forEach(c -> customerDataList.add(toData(c)));

		return customerDataList;
	}

}
